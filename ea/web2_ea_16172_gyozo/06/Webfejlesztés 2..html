Adatok módosítása és kirajzolás

var fr = 0;
var startTime = new Date();

var frame = 0;
var numberOfFrames = 16;

function update() {
  //FPS
  fr++;
  document.getElementById('fps').innerHTML = Math.floor(fr * 1000 / (new Date() - startTime));
  //Frame
  frame = (frame + 1) % numberOfFrames;
}

function draw() {
  ctx.fillStyle = '#000000';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  ctx.drawImage(img,
    (frame % 4) * 128, Math.floor(frame / 4) * 128, 127, 127,
    canvas.width / 2 - 64, canvas.height / 2 - 64, 128, 128);
}

Idő alapú animáció

var frame = 0;
var numberOfFrames = 16;
var lastTime = Date.now();
var cumDelta = 0;
var msPerFrame = 30;

function update() {
    //FPS
    //...
    //frame
    var most = Date.now();
    var delta = most - lastTime;
    if (cumDelta + delta > msPerFrame) {
        cumDelta = 0;
        frame = (frame + 1) % numberOfFrames;
    } else {
        cumDelta += delta;
    }
    lastTime = most;
}

var n;

function init(_n) {
  n = _n;
  map = _clearMap(n);
}

function increaseCell(x, y) {
  map[y][x] += 1;
}

// Belső függvények (nem szolgál külső meghívásra)
function _clearMap(n) {
  var map = [];
  for (var i = 0; i < n; i++) {
    map[i] = [];
    for (var j = 0; j < n; j++) {
      map[i].push(0);
    }
  }
  return map;
}

HTML generáló

function genTable(map) {
  //1. variáció: naív
  var s = '<table>';
  for (var i = 0; i < map.length; i++) {
    s += '<tr>';
    var row = map[i];
    for (var j = 0; j < row.length; j++) {
      var cell = row[j];
      s += '<td>' + cell + '</td>';
    }
    s += '</tr>';
  }
  s += '</table>';
  return s;
}

HTML generáló

function genTable(map) {
  //2. variáció: map + join
  return `
    <table>
      ${map.map(row => `
        <tr> 
          ${row.map(cell => `
            <td>${cell}</td>
          `).join('')} 
        </tr>
      `).join('')}
    </table>
  `;
}
